    override fun onTextChanged(s: CharSequence?, start: Int, before: Int, count: Int) {
        val item_text: String = s.toString()
        val rv_item = findViewById<RecyclerView>(R.id.rv_item)
        val rv_item2 = findViewById<RecyclerView>(R.id.rv_item2)
        val rv_item3 = findViewById<RecyclerView>(R.id.rv_item3)
        val rv_item4 = findViewById<RecyclerView>(R.id.rv_item4)
        val rv_item5 = findViewById<RecyclerView>(R.id.rv_item5)
        val rv_item6 = findViewById<RecyclerView>(R.id.rv_item6)

        rv_item.visibility = View.VISIBLE
        rv_item2.visibility = View.GONE
        rv_item3.visibility = View.GONE
        rv_item4.visibility = View.GONE
        rv_item5.visibility = View.GONE
        rv_item6.visibility = View.GONE

        var ans: Int = 0;

        if (ans == 0) {
            ItemList.clear()
            (rv_item.adapter as Item_Adapter).notifyDataSetChanged()

        }

        for(i in falselist.indices) {
            if (falselist[i].name1.indexOf(item_text) != -1 && item_text.length > 0) {
                var named = falselist[i].name1.filter { it.startsWith(item_text) }
                if (falselist[i].name1 == named)                                                            //falselist[0].name1 == falselist[0].name1.filter { startsWith(item_text) }) 이름과 필터로 걸러진 이름이 같으면.
                {
                    ItemList.add(falselist[i])
                    (rv_item.adapter as Item_Adapter).notifyDataSetChanged()
                }
            }
        }


    if(item_text.length == 0)
    {
        rv_item.visibility = View.GONE
        rv_item2.visibility = View.GONE
        rv_item3.visibility = View.GONE
        rv_item4.visibility = View.GONE
        rv_item5.visibility = View.GONE
        rv_item6.visibility = View.VISIBLE
    }
}
